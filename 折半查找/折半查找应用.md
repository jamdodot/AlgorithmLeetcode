### 875. 爱吃香蕉的珂珂
```cpp
class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        int maxInt=-1;
        for(int j:piles){//确定最大值
            if(j>maxInt)
               maxInt=j;
        }
        int left=1;int right=maxInt+1;//要找的数的范围
        int mid;
        while(left<right){
             mid=left+(right-left)/2;
            if(canfind(piles,mid,h)){
                right=mid;//虽然满足但不是最小
            }else{
            left=mid+1;
            }
        }
       
        return left;
    }

    bool canfind(vector<int>& piles,int speed,int h){
        int count=0;
        for(int i:piles){//forEach语法
                
            count+=(i/speed)+((i%speed>0)?1:0);//一定要加括号，不然恒等于1

        }
        return count<=h;
    }
};

```


### 1011. 在 D 天内送达包裹的能力
```cpp
class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int sum=0;
        for(int s:weights){
                sum+=s;
        }
        int left=sum/D;
        int right=sum+1;
        while(left<right){
            int mid=left+(right-left)/2;
            if(canfind(weights,mid,D)){
                right=mid;//同上
            }else{
                left=mid+1;
            }
        }
        return left;

    }
    bool canfind(vector<int>& weights,int cap,int D){
        int i=0;
        for(int d=1;d<=D;d++){
            int maxcap=cap;//避免后续循环无法获得cap
            while((maxcap-=weights[i])>=0){//注意这种写法
                i++;
                if(i==weights.size()){//妙啊
                    return true;
                }
            }
        }
        return false;
    }
};
```
